using Rhino;
using Rhino.FileIO;

namespace RebusConnector.Helpers;

/// <summary>
/// Exports the current Rhino document (or a layer-filtered subset) to a
/// temporary .3dm file that can be uploaded to 3DConvert.
///
/// The full document is always written — layer filtering is handled server-side
/// via the <c>included_layers</c> parameter, which keeps this helper simple and
/// ensures all material/texture definitions are present in the file.
/// </summary>
public static class RhinoExport
{
    /// <summary>
    /// Writes the active Rhino document to a temp .3dm file.
    /// Returns the path to the temp file (caller is responsible for deletion).
    /// Throws if the write fails.
    /// </summary>
    /// <param name="doc">The active Rhino document.</param>
    /// <param name="selectionOnly">
    /// When <c>true</c>, only the objects currently selected in the active viewport
    /// are written. Throws <see cref="InvalidOperationException"/> if nothing is selected.
    /// </param>
    public static string ExportCurrentDocTo3dm(RhinoDoc doc, bool selectionOnly = false)
    {
        var tempDir = Path.Combine(Path.GetTempPath(), "rebus_connector");
        Directory.CreateDirectory(tempDir);

        // Use the document name (or "untitled") so the 3DConvert service can
        // show a friendly filename in the Speckle commit message.
        var baseName = string.IsNullOrWhiteSpace(doc.Name)
            ? "untitled"
            : SanitizeFileName(Path.GetFileNameWithoutExtension(doc.Name));

        var tempPath = Path.Combine(tempDir, $"{baseName}_{Guid.NewGuid():N}.3dm");

        if (selectionOnly)
        {
            var selected = doc.Objects.GetSelectedObjects(false, false).ToArray();
            if (selected.Length == 0)
                throw new InvalidOperationException(
                    "No objects are selected in the viewport.\n" +
                    "Select the objects you want to export, then try again.");

            var options = new FileWriteOptions
            {
                FileVersion              = 8,
                WriteSelectedObjectsOnly = true,
            };

            if (!doc.WriteFile(tempPath, options))
                throw new InvalidOperationException(
                    "Rhino could not write the selected objects to a temporary file.");

            return tempPath;
        }

        var fullOptions = new FileWriteOptions
        {
            FileVersion              = 8,
            WriteSelectedObjectsOnly = false,
        };

        if (!doc.WriteFile(tempPath, fullOptions))
            throw new InvalidOperationException(
                "Rhino could not write the document to a temporary file.\n" +
                "Make sure the document has geometry and is not in a broken state.");

        return tempPath;
    }

    /// <summary>
    /// Returns all layer full-paths in the current document.
    /// Full-path uses '::' as the separator, matching Speckle / 3DConvert conventions.
    /// </summary>
    public static IReadOnlyList<(string FullPath, string DisplayName, int Depth)> GetLayerTree(RhinoDoc doc)
    {
        var result = new List<(string, string, int)>();

        foreach (var layer in doc.Layers)
        {
            if (layer.IsDeleted) continue;

            // Layer.FullPath returns "Parent::Child::GrandChild"
            var fullPath    = layer.FullPath;
            var depth       = fullPath.Split(new[] { "::" }, StringSplitOptions.None).Length - 1;
            var displayName = layer.Name;

            result.Add((fullPath, displayName, depth));
        }

        return result;
    }

    private static string SanitizeFileName(string name)
    {
        var invalid = Path.GetInvalidFileNameChars();
        var safe    = new string(name.Select(c => invalid.Contains(c) ? '_' : c).ToArray());
        return string.IsNullOrWhiteSpace(safe) ? "untitled" : safe;
    }
}

using Rhino;
using Rhino.DocObjects;
using Rhino.FileIO;
using Rhino.Geometry;

namespace RebusConnector.Helpers;

/// <summary>
/// Imports geometry from a .3dm file into the active Rhino document,
/// recreating the layer hierarchy and placing objects on matching layers.
/// Existing geometry on those layers is NOT removed — new objects are added
/// alongside existing ones.
/// </summary>
public static class RhinoImport
{
    /// <summary>
    /// Reads <paramref name="path"/> and adds all objects to <paramref name="doc"/>.
    /// Returns the count of imported objects.
    /// </summary>
    public static int ImportFile3dm(string path, RhinoDoc doc)
    {
        var file3dm = File3dm.Read(path)
            ?? throw new InvalidOperationException($"Cannot read 3DM file: {path}");

        // Build a full-path → doc-layer-index map as we go, so parent layers are
        // only created once even if many objects share them.
        var layerIndexCache = new Dictionary<string, int>(StringComparer.OrdinalIgnoreCase);

        int imported = 0;

        foreach (var obj in file3dm.Objects)
        {
            var geo = obj.Geometry;
            if (geo == null) continue;

            // Resolve the layer from the source file
            var srcLayer = obj.Attributes.LayerIndex >= 0 &&
                           obj.Attributes.LayerIndex < file3dm.AllLayers.Count
                           ? file3dm.AllLayers.FindIndex(obj.Attributes.LayerIndex)
                           : null;

            var fullPath   = srcLayer?.FullPath ?? "Received";
            var layerColor = srcLayer?.Color;

            var attrs = new ObjectAttributes
            {
                LayerIndex = FindOrCreateLayer(doc, fullPath, layerColor, layerIndexCache),
            };

            doc.Objects.Add(geo, attrs);
            imported++;
        }

        if (imported > 0)
            doc.Views.Redraw();

        return imported;
    }

    // ── Layer resolution ──────────────────────────────────────────────────────

    private static int FindOrCreateLayer(
        RhinoDoc doc,
        string fullPath,
        System.Drawing.Color? color,
        Dictionary<string, int> cache)
    {
        if (cache.TryGetValue(fullPath, out var cached))
            return cached;

        // Check if it already exists in the document
        var existing = doc.Layers.FindByFullPath(fullPath, -1);
        if (existing >= 0)
        {
            cache[fullPath] = existing;
            return existing;
        }

        // Build the hierarchy segment by segment (Rhino uses "::" as separator)
        var parts       = fullPath.Split(new[] { "::" }, StringSplitOptions.RemoveEmptyEntries);
        int parentIndex = -1;
        string accumulated = "";

        for (int i = 0; i < parts.Length; i++)
        {
            accumulated = i == 0 ? parts[i] : $"{accumulated}::{parts[i]}";

            if (cache.TryGetValue(accumulated, out var ci))
            {
                parentIndex = ci;
                continue;
            }

            int idx = doc.Layers.FindByFullPath(accumulated, -1);
            if (idx < 0)
            {
                var layer = new Layer { Name = parts[i] };
                if (parentIndex >= 0)
                    layer.ParentLayerId = doc.Layers[parentIndex].Id;
                if (i == parts.Length - 1 && color.HasValue)
                    layer.Color = color.Value;

                idx = doc.Layers.Add(layer);
            }

            cache[accumulated] = idx;
            parentIndex = idx;
        }

        return parentIndex;
    }
}

    // ── Navigation ────────────────────────────────────────────────────────────

    private void NavigateTo(NavState state, ConnectorCard? card = null)
    {
        _navHistory.Push(_navState);
        _navState    = state;
        if (card != null) _editingCard = card;
        SwapContent();
    }

    private void NavigateBack()
    {
        _navState = _navHistory.Count > 0 ? _navHistory.Pop() : NavState.Home;
        SwapContent();
    }

    private void GoHome()
    {
        _navHistory.Clear();
        _navState = NavState.Home;
        SwapContent();
    }

    private void SwapContent()
    {
        if (_contentPanel == null) return;
        Application.Instance.Invoke(() =>
        {
            TrimThemedLists();
            _contentPanel.Content = _navState switch
            {
                NavState.CardConfig    => BuildCardConfigScreen(_editingCard!),
                NavState.SelectProject => BuildSelectProjectScreen(_editingCard!),
                NavState.SelectModel   => BuildSelectModelScreen(_editingCard!),
                _                     => BuildHomeScreen(),
            };
        });
    }

    private void TrimThemedLists()
    {
        if (_themedPanels.Count > _baseThemedPanelsCount)
            _themedPanels.RemoveRange(_baseThemedPanelsCount, _themedPanels.Count - _baseThemedPanelsCount);
        if (_themedLabels.Count > _baseThemedLabelsCount)
            _themedLabels.RemoveRange(_baseThemedLabelsCount, _themedLabels.Count - _baseThemedLabelsCount);
    }

    private void RebuildCardList() => Application.Instance.Invoke(() =>
    {
        if (_navState == NavState.Home)
            SwapContent();
        ManageRhinoSelectionSubscription();
        ManageRhinoModifiedSubscription();
    });

                            {
                                new StackLayoutItem(successLbl, true),
                                new StackLayoutItem(setMsgBtn,  false),
                                new StackLayoutItem(viewBtn,    false),
                            }
                        };
                    }
                }
                else // receive card
                {
                    if (card.LastReceivedAt == null)
                    {
                        var lbl = new Label { Text = "Not loaded yet", Font = Av(9.5f), TextColor = TextLow, VerticalAlignment = VerticalAlignment.Center };

    // ── Send config body ──────────────────────────────────────────────────────

    private void BuildSendConfigBody(ConnectorCard card, CardState state, StackLayout stack)
    {
        // ── Mode dropdown ─────────────────────────────────────────────────────

        var modeDropDown = new DropDown();
        StyleInputBox(modeDropDown);
        modeDropDown.DataStore     = new[] { "Selection", "Layers", "All layers" };
        modeDropDown.SelectedIndex = card.LayerMode switch
        {
            "selection" => 0,
            "select"    => 1,
            _           => 2,
        };

        stack.Items.Add(new StackLayoutItem(ContentPad(modeDropDown, 8), HorizontalAlignment.Stretch));
        stack.Items.Add(new StackLayoutItem(Divider(), HorizontalAlignment.Stretch));

        // ── Dynamic content area ──────────────────────────────────────────────

        var contentArea = TrackPanel(new Panel { BackgroundColor = BgMain }, () => BgMain);

        // Layer checklist controls
        var layerStack  = new StackLayout { Spacing = 0, HorizontalContentAlignment = HorizontalAlignment.Stretch };
        var layerSearch = new TextBox { PlaceholderText = "Search layers…" };
        StyleInputBox(layerSearch);
        var layerScroll = new Scrollable
        {
            Height             = 160,
            ExpandContentWidth = true,
            BackgroundColor    = BgMain,
            Content            = layerStack,
        };

        // Labels for selection / all-layers count
        var selCountLabel = TrackLabel(new Label { Font = Av(10f), TextColor = TextMid }, () => TextMid);
        var allCountLabel = TrackLabel(new Label { Font = Av(10f), TextColor = TextMid }, () => TextMid);

        void RefreshLayers()
        {
            var doc = RhinoDoc.ActiveDoc;
            if (doc == null) return;

            state.Layers.Clear();
            layerStack.Items.Clear();

            var siblingLayers = _cardStore.GetSiblingLayers(card.Id, card.ModelId);
            var rows          = RhinoExport.GetLayerTree(doc).ToList();
            var filter        = layerSearch.Text.Trim().ToLowerInvariant();

            for (int i = 0; i < rows.Count; i++)
            {
                var (fullPath, displayName, depth) = rows[i];
                if (!string.IsNullOrEmpty(filter) &&
                    !displayName.ToLowerInvariant().Contains(filter))
                    continue;

                var item = new LayerItem { FullPath = fullPath, DisplayName = displayName, Depth = depth };
                item.IsSelected = card.IncludedLayers.Count == 0 || card.IncludedLayers.Contains(fullPath);
                state.Layers.Add(item);

                var cb = new CheckBox
                {
                    Text      = displayName,
                    Checked   = item.IsSelected,
                    Font      = Av(10.5f),
                    TextColor = TextMid,
                };

                var captured = item;
                cb.CheckedChanged += (_, _) =>
                {
                    captured.IsSelected = cb.Checked == true;
                    card.IncludedLayers = state.Layers.Where(l => l.IsSelected).Select(l => l.FullPath).ToList();
                    _cardStore.UpdateQuiet(card);
                };

                Control rowContent;
                if (siblingLayers.Contains(fullPath))
                {
                    var dot = new Label
                    {
                        Text              = "●",
                        TextColor         = ColGreen,
                        Font              = Av(10f),
                        VerticalAlignment = VerticalAlignment.Center,
                        ToolTip           = "Already in another publish group for this model",
                    };
                    rowContent = new StackLayout
                    {
                        Orientation              = Orientation.Horizontal,
                        Spacing                  = 4,
                        VerticalContentAlignment = VerticalAlignment.Center,
                        Items                    = { dot, cb },
                    };
                }
                else
                {
                    rowContent = cb;
                }

                layerStack.Items.Add(new StackLayoutItem(new Panel
                {
                    Padding         = new Padding(depth * 14, 3, 0, 3),
                    BackgroundColor = i % 2 == 0 ? BgCard : BgMain,
                    Content         = rowContent,
                }, HorizontalAlignment.Stretch));
            }
        }

        void RefreshObjectCount()
        {
            var summary = GetObjectSummary(card);
            Application.Instance.Invoke(() =>
            {
                selCountLabel.Text = summary;
                allCountLabel.Text = summary;
            });
        }

        state.OnSelectionChanged = RefreshObjectCount;

        void UpdateContentArea()
        {
            card.LayerMode = modeDropDown.SelectedIndex switch
            {
                0 => "selection",
                1 => "select",
                _ => "all",
            };
            _cardStore.UpdateQuiet(card);

            switch (modeDropDown.SelectedIndex)
            {
                case 0:
                    RefreshObjectCount();
                    contentArea.Content = ContentPad(selCountLabel, 10);
                    break;
                case 1:
                    RefreshLayers();
                    contentArea.Content = ContentPad(new StackLayout
                    {
                        Spacing                    = 6,
                        HorizontalContentAlignment = HorizontalAlignment.Stretch,
                        Items                      = { layerSearch, layerScroll }
                    }, 8);
                    break;
                default:
                    RefreshObjectCount();
                    contentArea.Content = ContentPad(allCountLabel, 10);
                    break;
            }
        }

        modeDropDown.SelectedIndexChanged += (_, _) => UpdateContentArea();
        layerSearch.TextChanged           += (_, _) => RefreshLayers();
        UpdateContentArea();

        stack.Items.Add(new StackLayoutItem(contentArea, HorizontalAlignment.Stretch));
        stack.Items.Add(new StackLayoutItem(Divider(),   HorizontalAlignment.Stretch));

        // ── Settings section ──────────────────────────────────────────────────

        stack.Items.Add(new StackLayoutItem(SectionHeader("SETTINGS"), HorizontalAlignment.Stretch));
        stack.Items.Add(new StackLayoutItem(Divider(), HorizontalAlignment.Stretch));

        // Project row
        var projNameLbl = TrackLabel(new Label
        {
            Text              = string.IsNullOrEmpty(card.ProjectName) ? "Select a project…" : card.ProjectName,
            Font              = Av(11f),
            VerticalAlignment = VerticalAlignment.Center,
        }, () => string.IsNullOrEmpty(card.ProjectName) ? TextLow : TextHigh);

        var projectRow = BuildSettingsRow("Project", projNameLbl);
        projectRow.MouseDown += (_, _) =>
        {
            if (state.Projects.Count == 0 && _graphql != null)
                _ = LoadProjectsAsync(state);
            NavigateTo(NavState.SelectProject, card);
        };

        stack.Items.Add(new StackLayoutItem(projectRow, HorizontalAlignment.Stretch));
        stack.Items.Add(new StackLayoutItem(Divider(),  HorizontalAlignment.Stretch));

        // Model row
        var modelNameLbl = TrackLabel(new Label
        {
            Text              = string.IsNullOrEmpty(card.ModelName) ? "Select a model…" : card.ModelName,
            Font              = Av(11f),
            VerticalAlignment = VerticalAlignment.Center,
        }, () => string.IsNullOrEmpty(card.ModelName) ? TextLow : TextHigh);

        var modelRow = BuildSettingsRow("Model", modelNameLbl);
        modelRow.MouseDown += (_, _) =>
        {
            if (!string.IsNullOrEmpty(card.ProjectId) && state.Models.Count == 0 && _graphql != null)
                _ = LoadModelsAsync(state, card);
            NavigateTo(NavState.SelectModel, card);
        };

        stack.Items.Add(new StackLayoutItem(modelRow, HorizontalAlignment.Stretch));
        stack.Items.Add(new StackLayoutItem(Divider(), HorizontalAlignment.Stretch));

        // Message
        var messageBox = new TextBox { PlaceholderText = "Optional version message", Text = state.PendingMessage };
        StyleInputBox(messageBox);
        messageBox.TextChanged += (_, _) => state.PendingMessage = messageBox.Text;

        stack.Items.Add(new StackLayoutItem(TrackPanel(new Panel
        {
            BackgroundColor = BgMain,
            Padding         = new Padding(16, 8),
            Content         = new StackLayout
            {
                Spacing                    = 4,
                HorizontalContentAlignment = HorizontalAlignment.Stretch,
                Items                      = { SectionLabel("MESSAGE"), messageBox }
            }
        }, () => BgMain), HorizontalAlignment.Stretch));
        stack.Items.Add(new StackLayoutItem(Divider(), HorizontalAlignment.Stretch));

#if INCLUDE_DEV_TARGET
        var (prodRadio, prodRadioPair) = RadioPair("Production");
        var (devRadio,  devRadioPair)  = RadioPair("Development", prodRadio);
        prodRadio.Checked = card.Target != "dev";
        devRadio.Checked  = card.Target == "dev";
        prodRadio.CheckedChanged += (_, _) => { card.Target = "prod"; _cardStore.UpdateQuiet(card); };
        devRadio.CheckedChanged  += (_, _) => { card.Target = "dev";  _cardStore.UpdateQuiet(card); };

        stack.Items.Add(new StackLayoutItem(SectionHeader("TARGET"), HorizontalAlignment.Stretch));
        stack.Items.Add(new StackLayoutItem(Divider(), HorizontalAlignment.Stretch));
        stack.Items.Add(new StackLayoutItem(ContentPad(new StackLayout
        {
            Orientation = Orientation.Horizontal,
            Spacing     = 20,
            Items       = { prodRadioPair, devRadioPair },
        }), HorizontalAlignment.Stretch));
        stack.Items.Add(new StackLayoutItem(Divider(), HorizontalAlignment.Stretch));
#endif

        // ── Publish button + config status ────────────────────────────────────

        var publishBtn    = new Button { Text = "Publish" };
        var configStatus  = TrackPanel(new Panel { BackgroundColor = BgMain }, () => BgMain);
        StylePrimaryButton(publishBtn);

        void RefreshConfigStatus()
        {
            Application.Instance.Invoke(() =>
            {
                if (state.IsBusy)
                {
                    publishBtn.Enabled = false;
                    var bar = new ProgressBar { Indeterminate = true };
                    var lbl = new Label { Text = state.BusyText, Font = Av(10f), TextColor = TextMid };
                    configStatus.Content = ContentPad(new StackLayout
                    {
                        Spacing                    = 4,

    // ── Publish flow ──────────────────────────────────────────────────────────

    private async Task DoPublish(ConnectorCard card)
    {
        var state = GetCardState(card.Id);

        if (string.IsNullOrEmpty(_token))
        { SetCardError(card, "Not logged in."); return; }

        if (string.IsNullOrEmpty(card.ProjectId))
        { SetCardError(card, "Select a project first."); return; }

        if (string.IsNullOrEmpty(card.ModelName))
        { SetCardError(card, "Select or enter a model name."); return; }

        var doc = RhinoDoc.ActiveDoc;
        if (doc == null) { SetCardError(card, "No active Rhino document."); return; }

        bool selectionOnly = card.LayerMode == "selection";
        IEnumerable<string>? includedLayers = null;
        if (card.LayerMode == "select")
            includedLayers = state.Layers.Where(l => l.IsSelected).Select(l => l.FullPath).ToList();

        SetCardBusy(card, "Exporting document…");

        string? tempFile = null;
        try
        {
            tempFile = await Task.Run(() => RhinoExport.ExportCurrentDocTo3dm(doc, selectionOnly));

            SetCardBusy(card, "Uploading…");
            var client = new ConvertApiClient(TokenStore.ConvertServerUrl, _token);
            var jobId  = await client.SubmitAsync(tempFile, card.ProjectId, card.ModelName,
                state.PendingMessage, includedLayers, card.Target);

            var progress = new Progress<string>(msg => SetCardBusy(card, msg));
            var job = await client.WaitForCompletionAsync(jobId, progress);

            if (job.Succeeded && !string.IsNullOrEmpty(job.SpeckleUrl))
            {
                state.LastResultUrl = job.SpeckleUrl;
                state.LastError     = null;
                card.LastVersionId  = job.SpeckleUrl;
                card.LastSentAt     = DateTime.UtcNow;
                _cardStore.UpdateQuiet(card);
                _modifiedSincePublish.Remove(card.Id);

                // Try to get model ID if it was a new model
                if (_graphql != null && string.IsNullOrEmpty(card.ModelId))
                {
                    try
                    {
                        var mdls = await _graphql.ListModelsAsync(card.ProjectId);
                        state.Models = mdls;
                        var idx = mdls.FindIndex(m =>
                            string.Equals(m.Name, card.ModelName, StringComparison.OrdinalIgnoreCase));
                        if (idx >= 0)
                        {
                            card.ModelId   = mdls[idx].Id;
                            card.ModelName = mdls[idx].Name;
                            _cardStore.UpdateQuiet(card);
                        }
                    }
                    catch { }
                }

                ClearCardBusy(card);
            }
            else
            {
                SetCardError(card, $"Conversion failed: {job.Error ?? "unknown error"}");
            }
        }
        catch (OperationCanceledException)
        {
            SetCardError(card, "Publish was cancelled.");
        }
        catch (System.Net.Http.HttpRequestException ex)
            when (ex.StatusCode == System.Net.HttpStatusCode.Unauthorized)
        {
            SetCardError(card, "Session expired — please log out and log back in.");
        }
        catch (Exception ex)
        {
            SetCardError(card, ex.Message);
        }
        finally
        {
            ClearCardBusy(card);
            if (tempFile != null) try { File.Delete(tempFile); } catch { }
        }
    }

    // ── Load flow ─────────────────────────────────────────────────────────────

    private async Task DoLoad(ConnectorCard card)
    {
        if (string.IsNullOrEmpty(_token))
        { SetCardError(card, "Not logged in."); return; }

        if (string.IsNullOrEmpty(card.ProjectId))
        { SetCardError(card, "Select a project first."); return; }

        if (string.IsNullOrEmpty(card.ModelId))
        { SetCardError(card, "Select a model first."); return; }

        var doc = RhinoDoc.ActiveDoc;
        if (doc == null) { SetCardError(card, "No active Rhino document."); return; }

        // card.PinnedVersionId is kept in sync by the config screen radio/combo;
        // null means "latest version".
        string? versionId = string.IsNullOrEmpty(card.PinnedVersionId) ? null : card.PinnedVersionId;

        SetCardBusy(card, "Receiving from Speckle…");

        string? tempFile = null;
        try
        {
            var client = new ConvertApiClient(TokenStore.ConvertServerUrl, _token);
            tempFile   = await client.ReceiveAsync(card.ProjectId, card.ModelId, versionId, card.Target);

            SetCardBusy(card, "Importing geometry…");
            var count = await Task.Run(() => RhinoImport.ImportFile3dm(tempFile, doc));

            card.LastReceivedAt        = DateTime.UtcNow;
            card.LastReceivedVersionId = versionId ?? "latest";
            GetCardState(card.Id).LastError = null;
            _cardStore.UpdateQuiet(card);

            ClearCardBusy(card);
        }
        catch (OperationCanceledException) { SetCardError(card, "Load was cancelled."); }
        catch (Exception ex)               { SetCardError(card, ex.Message); }
        finally
        {
            ClearCardBusy(card);
            if (tempFile != null) try { File.Delete(tempFile); } catch { }
        }
    }

    private static string GetCardTitle(ConnectorCard card)

    private string GetObjectSummary(ConnectorCard card)
    {
        if (card.Type == CardType.Receive)
        {
            if (string.IsNullOrEmpty(card.ProjectName)) return "(no project selected)";
            return string.IsNullOrEmpty(card.ModelName)
                ? card.ProjectName
                : $"{card.ProjectName} › {card.ModelName}";
        }

        var doc = RhinoDoc.ActiveDoc;
        if (doc == null) return "";

        switch (card.LayerMode)
        {
            case "selection":
            {
                var objs = doc.Objects.GetSelectedObjects(false, false).ToList();
                if (objs.Count == 0) return "Selection · no objects selected";
                var groups = GetTypeGroups(objs);
                var layers = objs
                    .Select(o => o.Attributes.LayerIndex >= 0 && o.Attributes.LayerIndex < doc.Layers.Count
                        ? doc.Layers[o.Attributes.LayerIndex]?.FullPath ?? ""
                        : "")
                    .Distinct().Count(p => !string.IsNullOrEmpty(p));
                return $"Selection: {string.Join(" · ", groups)} from {layers} {(layers == 1 ? "layer" : "layers")}";
            }
            case "select":
            {
                if (card.IncludedLayers.Count == 0) return "Layers: none selected";
                var names  = card.IncludedLayers.Select(p => p.Split("::").Last()).Take(3).ToList();
                var suffix = card.IncludedLayers.Count > 3 ? $" +{card.IncludedLayers.Count - 3}" : "";
                return $"Layers: {string.Join(", ", names)}{suffix}";
            }
            default:
            {
                var count = doc.Objects.Count(o => !o.IsHidden && o.IsValid);
                return $"All layers · {count} {(count == 1 ? "object" : "objects")}";
            }
        }
    }

    private static IEnumerable<string> GetTypeGroups(
        IEnumerable<Rhino.DocObjects.RhinoObject> objects)
    {
        return objects
            .GroupBy(o => o.ObjectType switch
            {
                Rhino.DocObjects.ObjectType.Mesh      => "Mesh",
                Rhino.DocObjects.ObjectType.Brep      => "Brep",
                Rhino.DocObjects.ObjectType.Extrusion => "Brep",
                Rhino.DocObjects.ObjectType.Curve     => "Curve",
                Rhino.DocObjects.ObjectType.Point     => "Point",
                _                                     => "Other"
            })
            .OrderByDescending(g => g.Count())
            .Select(g => $"{g.Count()} {g.Key}");
    }

                ShowLoginForm();
                return;

            _browserSection.Visible = false;
            TokenStore.SaveToken(serverUrl, token);
            await InitialiseLoggedInAsync(serverUrl, token);
        }
        catch (OperationCanceledException) { ShowLoginForm(); }
        catch (Exception ex) { SetLoginError(ex.Message); ShowLoginForm(); }

    // ── Rhino selection subscription ──────────────────────────────────────────

    private void ManageRhinoSelectionSubscription()
    {
        bool anyInSelectionMode = _cardStore.Cards
            .Any(c => c.Type == CardType.Send && c.LayerMode == "selection");

        if (anyInSelectionMode && !_rhinoSelectionSubscribed)
        {
            RhinoDoc.SelectObjects      += OnRhinoSelectionChanged;
            RhinoDoc.DeselectObjects    += OnRhinoSelectionChanged;
            RhinoDoc.DeselectAllObjects += OnRhinoDeselectAll;
            _rhinoSelectionSubscribed = true;
        }
        else if (!anyInSelectionMode && _rhinoSelectionSubscribed)
        {
            UnsubscribeRhinoSelectionEvents();
        }
    }

    private void OnRhinoSelectionChanged(object? sender, Rhino.DocObjects.RhinoObjectSelectionEventArgs e)
    {
        foreach (var s in _cardStates.Values)
            s.OnSelectionChanged?.Invoke();
    }

    private void OnRhinoDeselectAll(object? sender, Rhino.DocObjects.RhinoDeselectAllObjectsEventArgs e)
    {
        foreach (var s in _cardStates.Values)
            s.OnSelectionChanged?.Invoke();
    }

    private void UnsubscribeRhinoSelectionEvents()
    {
        if (_rhinoSelectionSubscribed)
        {
            RhinoDoc.SelectObjects      -= OnRhinoSelectionChanged;
            RhinoDoc.DeselectObjects    -= OnRhinoSelectionChanged;
            RhinoDoc.DeselectAllObjects -= OnRhinoDeselectAll;
            _rhinoSelectionSubscribed   = false;
        }
    }

    // ── Rhino document modification subscription (out-of-sync detection) ──────
    //
    // Subscribes to AddRhinoObject / DeleteRhinoObject so that geometry changes
    // mark published send cards as "out of sync with application".

    private void ManageRhinoModifiedSubscription()
    {
        bool anyPublished = _cardStore.Cards
            .Any(c => c.Type == CardType.Send && c.LastSentAt != null);

        if (anyPublished && !_rhinoModifiedSubscribed)
        {
            RhinoDoc.AddRhinoObject    += OnRhinoObjectAdded;
            RhinoDoc.DeleteRhinoObject += OnRhinoObjectDeleted;
            _rhinoModifiedSubscribed = true;
        }
        else if (!anyPublished && _rhinoModifiedSubscribed)
        {
            UnsubscribeRhinoModifiedEvents();
        }
    }

    private void OnRhinoObjectAdded(object? sender, Rhino.DocObjects.RhinoObjectEventArgs e)
        => MarkCardsOutOfSync();

    private void OnRhinoObjectDeleted(object? sender, Rhino.DocObjects.RhinoObjectEventArgs e)
        => MarkCardsOutOfSync();

    private void MarkCardsOutOfSync()
    {
        bool anyChanged = false;
        foreach (var card in _cardStore.Cards.Where(c => c.Type == CardType.Send && c.LastSentAt != null))
        {
            if (_modifiedSincePublish.Add(card.Id))
                anyChanged = true;
        }

        if (anyChanged)
        {
            foreach (var card in _cardStore.Cards.Where(c => _modifiedSincePublish.Contains(c.Id)))
            {
                var state = GetCardState(card.Id);
                Application.Instance.Invoke(() => state.RefreshStatusBar?.Invoke());
            }
        }
    }

    private void UnsubscribeRhinoModifiedEvents()
    {
        if (_rhinoModifiedSubscribed)
        {
            RhinoDoc.AddRhinoObject    -= OnRhinoObjectAdded;
            RhinoDoc.DeleteRhinoObject -= OnRhinoObjectDeleted;
            _rhinoModifiedSubscribed = false;
        }
    }

    protected override void OnUnLoad(EventArgs e)
    {
        UnsubscribeRhinoSelectionEvents();
        UnsubscribeRhinoModifiedEvents();
        _cardStore.Dispose();
        base.OnUnLoad(e);
    }
}

using System.Text.Json;
using Rhino;

namespace RebusConnector.Models;

/// <summary>
/// Persists connector cards as JSON in the active Rhino document's string table
/// (key "rebus_connector" / "cards") so they travel with the .3dm file.
/// Fires <see cref="CardsChanged"/> on load or any mutation.
/// </summary>
public class CardStore : IDisposable
{
    private const string Section = "rebus_connector";
    private const string Entry   = "cards";

    private static readonly JsonSerializerOptions _json = new()
    {
        WriteIndented        = false,
        PropertyNamingPolicy = null,
    };

    private List<ConnectorCard> _cards = [];

    /// <summary>Raised on the calling thread whenever the card list changes.</summary>
    public event Action? CardsChanged;

    public CardStore()
    {
        RhinoDoc.ActiveDocumentChanged += OnDocumentChanged;
        Load();
    }

    public void Dispose() =>
        RhinoDoc.ActiveDocumentChanged -= OnDocumentChanged;

    // ── Public API ────────────────────────────────────────────────────────────

    public IReadOnlyList<ConnectorCard> Cards => _cards;

    public void Add(ConnectorCard card)
    {
        _cards.Add(card);
        Save();
        CardsChanged?.Invoke();
    }

    public void Remove(string cardId)
    {
        _cards.RemoveAll(c => c.Id == cardId);
        Save();
        CardsChanged?.Invoke();
    }

    public void Update(ConnectorCard card)
    {
        var idx = _cards.FindIndex(c => c.Id == card.Id);
        if (idx >= 0) _cards[idx] = card;
        else          _cards.Add(card);
        Save();
        CardsChanged?.Invoke();
    }

    /// <summary>Persists card state without raising <see cref="CardsChanged"/>.</summary>
    public void UpdateQuiet(ConnectorCard card)
    {
        var idx = _cards.FindIndex(c => c.Id == card.Id);
        if (idx >= 0) _cards[idx] = card;
        else          _cards.Add(card);
        Save();
    }

    /// <summary>
    /// Returns the union of IncludedLayers from all OTHER send cards targeting
    /// the same model — used to show green indicators in the layer checklist.
    /// </summary>
    public HashSet<string> GetSiblingLayers(string cardId, string modelId) =>
        _cards
            .Where(c => c.Id != cardId
                     && c.Type == CardType.Send
                     && c.ModelId == modelId)
            .SelectMany(c => c.IncludedLayers)
            .ToHashSet(StringComparer.OrdinalIgnoreCase);

    // ── Persistence ───────────────────────────────────────────────────────────

    private void OnDocumentChanged(object? sender, DocumentEventArgs e)
    {
        Load();
        CardsChanged?.Invoke();
    }

    private void Load()
    {
        var doc = RhinoDoc.ActiveDoc;
        if (doc == null) { _cards = []; return; }

        var json = doc.Strings.GetValue(Section, Entry);
        if (string.IsNullOrWhiteSpace(json)) { _cards = []; return; }

        try   { _cards = JsonSerializer.Deserialize<List<ConnectorCard>>(json, _json) ?? []; }
        catch { _cards = []; }
    }

    private void Save()
    {
        var doc = RhinoDoc.ActiveDoc;
        if (doc == null) return;
        doc.Strings.SetString(Section, Entry, JsonSerializer.Serialize(_cards, _json));
    }
}

    /// <summary>"all" | "select" | "selection"</summary>
    public string       LayerMode      { get; set; } = "all";
    public List<string> IncludedLayers { get; set; } = [];
    public string?      LastVersionId  { get; set; }
    public DateTime?    LastSentAt     { get; set; }

    // ── Receive-specific ──────────────────────────────────────────────────────

    /// <summary>null = always use latest version</summary>
    public string?   PinnedVersionId       { get; set; }
    public DateTime? LastReceivedAt        { get; set; }
    public string?   LastReceivedVersionId { get; set; }

    // ── Display helpers ───────────────────────────────────────────────────────

    [JsonIgnore]
    public string CardTitle => Type == CardType.Send
        ? LayerMode switch
        {
            "all"       => "All layers",
            "selection" => "Selection",
            _           => IncludedLayers.Count > 0
                               ? string.Join(", ", IncludedLayers
                                   .Select(p => p.Split("::").Last())
                                   .Take(3))
                               : "Select layers"
        }
        : string.IsNullOrEmpty(ModelName) ? "Receive" : ModelName;

    [JsonIgnore]
    public string LastActionText => Type == CardType.Send
        ? (LastSentAt.HasValue     ? $"Sent {TimeAgo(LastSentAt.Value)}"     : "Never sent")
        : (LastReceivedAt.HasValue ? $"Received {TimeAgo(LastReceivedAt.Value)}" : "Never received");

    private static string TimeAgo(DateTime dt)
    {
        var span = DateTime.UtcNow - dt.ToUniversalTime();
        if (span.TotalSeconds < 60)  return "just now";
        if (span.TotalMinutes < 60)  return $"{(int)span.TotalMinutes}m ago";
        if (span.TotalHours   < 24)  return $"{(int)span.TotalHours}h ago";
        return $"{(int)span.TotalDays}d ago";
    }
}

# REBUS Portal Connector for Rhino

A native Rhino 8 plugin that connects Rhino directly to the REBUS Speckle deployment, using the REBUS 3DConvert conversion pipeline for high-fidelity geometry upload and download.

---

## Overview

The REBUS Portal Connector replicates the Speckle connector workflow but routes all geometry through the REBUS 3DConvert service, which preserves:

- Full PBR materials (diffuse, emissive, metallic, roughness, opacity)
- Texture maps uploaded as Speckle blobs
- Per-mesh texture coordinates
- Layer colours and layer hierarchy
- Named views (3DM files only)

Authentication is handled by your existing REBUS Speckle account — no separate API key is required.

---

## Features

### Card-based workflow

The panel uses a persistent **card** system. Each card represents a saved send or receive configuration and is stored in the Rhino document (travels with the `.3dm` file).

- **Send cards** — define what to push and where (project, model, layer selection)
- **Receive cards** — define what to pull from Speckle and import into Rhino
- Multiple cards per document — push different layer groups to the same model independently
- Cards persist across Rhino sessions

### Layer selection modes

Each send card supports three modes:

| Mode | Behaviour |
|---|---|
| All layers | Exports the entire document |
| Select layers | Exports only the checked layers (layer checklist shown) |
| Selection | Exports only the objects currently selected in the viewport |

### Layer status indicators

In **Select layers** mode, layers that are already included in another send card targeting the same model are highlighted with a green dot `●`. This lets you see at a glance which layers are already being managed by other cards, preventing accidental duplication.

### Receive / download

Receive cards pull a Speckle version back into Rhino:

- Geometry is placed on the original layer hierarchy (layers are created if they don't exist)
- Existing geometry is not removed — new objects are added alongside it
- Choose **Latest version** or pin a **Specific version**

### Collaborative workflow

Multiple people can work on the same Speckle project and model simultaneously:

```
Model A
├── Card: Layers 1–3   → pushed by User A
├── Card: Layers 4–5   → pushed by User B
└── Card: Selection    → pushed by User A (detail work)
```

When layer 2 changes, only the card covering layers 1–3 needs to be re-pushed.

### Themes

Light and dark mode with a toggle button. Theme preference is persisted per-installation.

### Build variants

| Variant | Target servers available | Installer name |
|---|---|---|
| Release (public) | Production only | `Rebus-Portal-Connector_vX.X.X_Release.exe` |
| Debug (internal) | Production + Development | `Rebus-Portal-Connector_vX.X.X_Debug.exe` |

---

## Requirements

- Rhino 8 (Windows)
- Access to the REBUS Speckle server (`speckle.rebus.industries`)
- A REBUS Speckle account

---

## Installation

### Via installer (recommended)

1. Download the latest `Rebus-Portal-Connector_vX.X.X_Release.exe` from [Releases](../../releases)
2. Close Rhino if it is open
3. Run the installer (no administrator rights required)
4. Open Rhino — the **REBUS Connector** panel is available under Panels menu

### Manual (from build output)

1. Build the project (see [Building from source](#building-from-source))
2. In Rhino, go to **Tools → Options → Plug-ins → Install**
3. Select `RebusConnector.rhp` from `bin\x64\Release\net7.0-windows\`

---

## Usage

### First-time login

1. Open the **REBUS Connector** panel (Panels menu or type `RebusConnector` in the command line)
2. Click **LOG IN TO SPECKLE**
3. Complete the OAuth login in the embedded browser panel
4. The panel shows your name and server URL when logged in

### Creating a send card

1. Click **+ SEND** in the action bar
2. Select or create a **Project**
3. Select or create a **Model** (branch)
4. Choose a layer mode: All layers / Select layers / Selection
5. Optionally add a **version message**
6. Click **SEND TO SPECKLE**

The card is saved automatically. Next time you open the file, the card remembers its configuration.

### Creating a receive card

1. Click **+ RECEIVE** in the action bar
2. Select a **Project** and **Model**
3. Choose **Latest version** or select a specific version
4. Click **RECEIVE FROM SPECKLE**

Geometry is imported onto matching layers in the active document.

### Re-sending after changes

Open the file, find the relevant send card, and click **SEND**. The card already has the correct project, model, and layer configuration from the previous send.

## Key Conventions

- **IP ranges:** Production services on `10.0.200.1–99`, dev/staging on `10.0.200.100–199`, proxy/infra on `10.0.200.250–254`.
- **Compose rule:** always `cd /opt/speckle && docker compose ...` or use `--project-directory`. Never `-f docker-compose.yml` alone — silently drops the override.
- **Caddy changes:** always update **both** proxies (Proxy1 on SRV01, Proxy2 on SRV02) and reload both.
- **Custom images — versioning:** Frontend is independently versioned from server/preview-service. Built on dev VM from `rebus-dev` branch, transferred to prod via `docker save | gzip`. See `speckle-dev/SETUP_NOTES.md` for rebuild instructions for each.
  - `speckle-frontend-2-rebus:v2.3.0` — iFrame embedding, V3 named views, camera geometry fix, hardcoded version string, **Layer/Model/Level/Type labels for Collections** (full collectionType-aware sidebar: layer, model, rhino layer/model, root, level, type, legacy `:Speckle.Core.Models.Layer`). Current on prod + dev.
  - `speckle-server-rebus:v2.1.0` — custom server build with `FF_SAVED_VIEWS_ENABLED` and related feature flags
  - `speckle-preview-service-rebus:v2.1.0` — no-op Camera converter ensures thumbnails are clean
- **Speckle — do not blindly enable `FF_WORKSPACES_MODULE_ENABLED` on prod:** Enabling workspaces on `speckle-server` requires the forked web app to tolerate workspace GraphQL/`workspace:read` cleanly; flipping it historically **broke login** (`speckle/VIEWER_CUSTOMISATION.md` §§5–6). Prefer the documented approach (frontend GraphQL + server authz tweaks for saved views without assuming every project lives in a workspace).
- **3DConvert hot-patch warning:** All changes since initial build are hot-patched into the running container. A full `docker build` is needed to bake them in — do this before any planned downtime.

## Development Tools

3DConvert is developed in **Cursor** (AI code editor) as well as Claude Cowork sessions. The 3DConvert repo has a `.cursorrules` file at its root with full architectural context — open the `3DConvert/` folder (or the cloned GitHub repo) as the Cursor project root.

**Workflow for 3DConvert changes:**
1. Edit source in Cursor (local clone or `D:\Documents\Claude\REBUS System\3DConvert\app\`)
2. Hot-patch to running container: `docker stop → docker cp → clear pyc → docker start`
3. When stable: full rebuild on prod VM (`DOCKER_BUILDKIT=0 docker build --no-cache -t 3dconvert:latest . && docker compose up -d`)
4. Commit on prod VM, relay-push to GitHub via dev VM bundle (see `3DConvert/SETUP_NOTES.md`)

**Critical for embedded IronPython (`_RHINO_SCRIPT_CONTENT` in `rhino_conv.py`):**
- Edit only inside the container's `rhino_conv.py` — PC01's `rh_convert.py` is overwritten every job
- All strings in the embedded script must use `'''` not `"""` (would close the outer `r"""..."""`)
- All paths using backslash must use raw strings: `r"C:\..."` or `\\`
- All .NET integer types (`System.Byte`, `System.Int32`) must be cast with `int()` before JSON serialization

### 2026-04-27 — `speckle.rebus.industries` 502s (resolved)
Root cause: `speckle-update` called `docker compose -f docker-compose.yml up -d`, suppressing auto-discovery of `docker-compose.override.yml` and dropping host port bindings. # Main Speckle app - HTTP only (local network, no TLS)
http://192.168.1.38 {

	# API and backend routes go directly to speckle-server
	@backend path /graphql /explorer /auth/* /objects/* /preview/* /api/* /static/*

	handle @backend {
		request_body {

		}
		reverse_proxy speckle-server:3000 {
			header_down -X-Frame-Options
			header_down -Content-Security-Policy
			header_down Content-Security-Policy "frame-ancestors *"
		}
	}

	# Everything else goes to the frontend
	handle {
		request_body {
			max_size 100MB
		}

## Resolved — `speckle-update` no longer strips host port bindings (fixed 2026-04-27)

**Original symptom:** after running `speckle-update`, the external proxy started returning
502 for speckle.rebus.industries. `docker ps` showed all 8 containers running, but
`ss -ltn` showed nothing listening on 80/443/3000/9000/9001.

**Root cause:** `/usr/local/bin/speckle-update` invoked
`sudo docker compose -f /opt/speckle/docker-compose.yml up -d`. Passing `-f` explicitly
suppressed Compose's auto-merge of `docker-compose.override.yml`, which is the file
that publishes the host port bindings.

**Fix:** the two compose invocations in `do_update()` now use
`--project-directory /opt/speckle` (no explicit `-f`), so Compose auto-discovers and
merges both files.

```bash
sudo docker compose --project-directory /opt/speckle pull
sudo docker compose --project-directory /opt/speckle up -d
```

The original script is preserved at `/usr/local/bin/speckle-update.bak.20260427` if
anything ever needs to be reverted or compared.

  use it. The `speckle-update` command checks that all 5 service images exist for a given
  X.Y.Z stable version before updating. GitHub releases are published before Docker images
  are ready, so the script uses Docker Hub as the source of truth, not GitHub.
- X-Frame-Options and Content-Security-Policy headers blocking iframe embedding are set
  by the Speckle app itself — the external reverse proxy should strip/replace them if
  iframe embedding is needed. **Currently the proxy Caddyfile on RB-DA2-Proxy1/2 does
  NOT include these header rewrites** (see "External Reverse Proxy Requirements" below).
  The original on-host `Caddyfile` snapshot in this directory shows the rules that were
  used pre-migration.
- Data is stored as bind mounts in `/opt/speckle/` (not Docker volumes), making backup
  and migration straightforward.

## External Reverse Proxy Requirements
The external proxy (Caddy on RB-DA2-Proxy1/2 LXCs) replicates the routing logic that
Caddy on this host previously handled.

### Path-based routing (current Proxy1 Caddyfile)
- `/graphql /api/* /auth/* /objects/* /preview/* /static/*` → `10.0.200.11:3000` (speckle-server)
- Everything else → `10.0.200.11:80` (speckle-frontend-2)

> The Proxy1 Caddyfile currently routes by path but does **not** include `/explorer` in
> the backend matcher — the historical on-host Caddyfile did. If the GraphQL Explorer
> stops working, add `/explorer` to the path matcher.
>
> Likewise the proxy does **not** currently expose `:9000` (MinIO S3) or `:9001` (console)
> publicly via the proxy. Direct LAN access only.

### Request body size limits (configured in old on-host Caddyfile, NOT in current proxy)
- Backend routes: 1000MB max
- Frontend routes: 100MB max

If large file uploads start failing with 413, this is why — the new proxy needs the
`request_body { max_size 1000MB }` directive added to the speckle.rebus.industries block.

### Headers (if iframe embedding is needed)
Strip `X-Frame-Options` and `Content-Security-Policy` from responses, replace with:
`Content-Security-Policy: frame-ancestors *`

Currently NOT applied at the new proxy. Add `header_down` directives to the
speckle.rebus.industries block in Proxy1/Proxy2 Caddyfiles if iframe embedding breaks.
